def load_config():
# 
    try:
        print("Reading execution config ....")
        with open(os.getcwd()+'/config/config.yml', 'r') as file:
            conf = yaml.load(file, Loader=yaml.FullLoader)
        return conf
    except Exception as e:
        print('Reading execution config failed: ')
        print("Error: {0}".format(str(e)))

create a database connection to a SQLite database :param db_file: :return:

def create_connection(db_file):
# 
    conn = None
    try:
        conn = sqlite3.connect(db_file)
        print(sqlite3.version)
    except Error as e:
        print(e)
    finally:
        if conn:
            conn.close()

create a table from the create_table_sql statement

def create_table(conn, create_table_sql):
# 
:param conn: Connection object
:param create_table_sql: a CREATE TABLE statement
:return:

    try:
        c = conn.cursor()
        c.execute(create_table_sql)
    except Error as e:
        print(e)

To fetch and parse urls :param param: :type param: url object :rtype: object :return: soup: parsed html object

def get_and_parse_url(param):
# 
    result = requests.get(param)
    soup = BeautifulSoup(result.text, 'html.parser')
    return soup
